# fish completion for forge
function __forge_path_is
    set -l expected $argv
    set -l tokens (commandline -opc)
    if test (count $tokens) -lt 1
        return 1
    end
    set -e tokens[1]
    set -l actual
    for token in $tokens
        if string match -qr '^-' -- $token
            continue
        end
        set actual $actual $token
    end
    if test (count $actual) -ne (count $expected)
        return 1
    end
    for idx in (seq (count $expected))
        if test "$actual[$idx]" != "$expected[$idx]"
            return 1
        end
    end
    return 0
end
